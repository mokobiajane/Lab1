40// File header for BMP format
41struct BITMAPFILEHEADER {
42 uint16_t bfType; // File type (should be 0x4D42 for BMP)
43 uint32_t bfSize; // Size of the file
44 uint16_t bfReserved1; // Reserved, must be 0
45 uint16_t bfReserved2; // Reserved, must be 0
46 uint32_t bfOffBits; // Offset from the start of the file to the pixel data
47};
48
49// Info header for BMP format
50struct BITMAPINFOHEADER {
51 uint32_t biSize; // Size of this header (40 bytes)
52 int32_t biWidth; // Image width
53 int32_t biHeight; // Image height (negative for top-down BMP)
54 uint16_t biPlanes; // Number of color planes (must be 1)
55 uint16_t biBitCount; // Number of bits per pixel (typically 24)
56 uint32_t biCompression; // Compression type (0 for uncompressed)
57 uint32_t biSizeImage; // Size of the image data (including padding)
58 int32_t biXPelsPerMeter; // Horizontal resolution (pixels per meter)
59 int32_t biYPelsPerMeter; // Vertical resolution (pixels per meter)
60 uint32_t biClrUsed; // Number of colors used (0 for default)
61 uint32_t biClrImportant; // Number of important colors (0 for all)
62};
